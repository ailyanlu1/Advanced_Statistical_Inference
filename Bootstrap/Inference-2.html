We’re creating a population that has gamma distribution with paramters r=5 and lambda=1/4

dist<-rgamma(500, 5, 1/4)
mean_dist<-mean(dist)

Sampling Distribution of Means

samp_mean<-rgamma(500, 500*5, 500*(1/4))
samp_mu<-mean(samp_mean)
samp_sd<-sd(samp_mean)

hist(samp_mean, col = "purple", breaks = 10)

Creating a random sample of size 200 from the population

rand_sample<-sample(dist, 200, replace = F)

mean_rand_sample<-mean(rand_sample)
sd_rand_sample<-sd(rand_sample)

hist(rand_sample, col = "lightgreen", main = "Histogram of Random Sample(size=200)")

Bootstrap Distribution

N<-10^5
my.boot<-numeric(N)

for(i in 1:N){
  x<-sample(rand_sample, 200, replace = T)
  my.boot[i]<-mean(x)
}
mean_boot<-mean(my.boot)
sd_boot<-sd(my.boot)

hist(my.boot, col = "Orange", axes = F)
par(new=T)
plot(density(my.boot), main = "", xlab = "", ylab = "", lwd=2)

Comparison table:

r1<-c("Population", mean_dist, sd(dist))
r2<-c("Sampling Distribution of Mean", mean_value=samp_mu, sd_value=samp_sd)
r3<-c("Sample (size=200)", mean_rand_sample, sd_rand_sample)
r4<-c("Bootstrap Distribution", mean_boot, sd_boot)

result<- as.data.frame(rbind(r1, r2, r3, r4))
names(result)<-c("", "Mean", "Standard Deviation")
